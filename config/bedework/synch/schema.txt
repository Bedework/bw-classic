
    create table bwsyn_ipinfo (
        bwsynipinfo_id int8 not null,
        bwsyn_order int4,
        bwsyn_addr varchar(255) not null,
        bwsyn_block char(1),
        primary key (bwsynipinfo_id, bwsyn_addr)
    );

    create table bwsynch_config (
        bwsyn_id int8 not null,
        bwsyn_seq int4 not null,
        bwsyn_sling_poolsv int4,
        bwsyn_sling_pooltout int8,
        bwsyn_missing_retries int4,
        bwsyn_callback_uri varchar(1000),
        bwsyn_tzs_uri varchar(1000),
        bwsyn_keystore varchar(1000),
        bwsyn_privKeys varchar(1000),
        bwsyn_pubKeys varchar(1000),
        primary key (bwsyn_id)
    );

    create table bwsynch_conn (
        bwsyn_id int8 not null,
        bwsyn_seq int4 not null,
        bwsyn_name varchar(250) not null unique,
        bwsyn_cname varchar(1000),
        bwsyn_readonly char(1) not null,
        bwsyn_trustLastmod char(1) not null,
        bwsyncnctr_id int8,
        primary key (bwsyn_id)
    );

    create table bwsynch_subs (
        bwsyn_id int8 not null,
        bwsyn_seq int4 not null,
        bwsyn_subid varchar(250) not null unique,
        bwsyn_owner varchar(500) not null,
        bwsyn_lrefresh varchar(20),
        bwsyn_errorct int4,
        bwsyn_missing char(1) not null,
        bwsyn_connectorid_a varchar(100),
        bwsyn_conn_props_a varchar(3000),
        bwsyn_connectorid_b varchar(100),
        bwsyn_conn_props_b varchar(3000),
        bwsyn_props varchar(3000),
        bwsyn_dir varchar(25) not null,
        bwsyn_mstr varchar(25) not null,
        primary key (bwsyn_id)
    );

    create table bwsynconn_properties (
        id int8 not null,
        bwsyn_pname varchar(200),
        bwsyn_pvalue text
    );

    alter table bwsyn_ipinfo 
        add constraint bwsynipinfo_fk 
        foreign key (bwsynipinfo_id) 
        references bwsynch_config;

    create index bwsynconnidx_name on bwsynch_conn (bwsyn_name);

    alter table bwsynch_conn 
        add constraint bwsynconn_fk 
        foreign key (bwsyncnctr_id) 
        references bwsynch_config;

    create index bwsynidx_subowner on bwsynch_subs (bwsyn_owner);

    create index bwsynidx_subid on bwsynch_subs (bwsyn_subid);

    alter table bwsynconn_properties 
        add constraint bwsynconn_pid_fk 
        foreign key (id) 
        references bwsynch_conn;

    create sequence hibernate_sequence;
